title: jQuery将表单序列化成一个Object对象的实例
---

```
(function($){ 
  $.fn.extend({ 
    serializeObject:function(){ 
      if(this.length>1){ 
        return false; 
      } 
      var arr=this.serializeArray(); 
      var obj=new Object; 
      $.each(arr,function(k,v){ 
        obj[v.name]=v.value; 
      }); 
      return obj; 
    } 
  }); 
})(jQuery); 

$(":button").click(function(){
       var test=$("form").serializeObject();
       alert(test.id);     
    });
```
